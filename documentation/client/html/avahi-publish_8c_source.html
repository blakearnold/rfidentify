00169 static void client_callback(AvahiClient *c, AvahiClientState state, AVAHI_GCC_UNUSED void * userdata) {
00170     assert(c);
00171 
00172     /* Called whenever the client or server state changes */
00173 
00174     switch (state) {
00175         case AVAHI_CLIENT_S_RUNNING:
00176 
00177             /* The server has startup successfully and registered its host
00178              * name on the network, so it's time to create our services */
00179             create_services(c);
00180             break;
00181 
00182         case AVAHI_CLIENT_FAILURE:
00183 
00184             fprintf(stderr, "Client failure: %s\n", avahi_strerror(avahi_client_errno(c)));
00185             avahi_simple_poll_quit(simple_poll);
00186 
00187             break;
00188 
00189         case AVAHI_CLIENT_S_COLLISION:
00190 
00191             /* Let's drop our registered services. When the server is back
00192              * in AVAHI_SERVER_RUNNING state we will register them
00193              * again with the new host name. */
00194 
00195         case AVAHI_CLIENT_S_REGISTERING:
00196 
00197             /* The server records are now being established. This
00198              * might be caused by a host name change. We need to wait
00199              * for our own records to register until the host name is
00200              * properly esatblished. */
00201 
00202             if (group)
00203                 avahi_entry_group_reset(group);
00204 
00205             break;
00206 
00207         case AVAHI_CLIENT_CONNECTING:
00208             ;
00209     }
00210 }
